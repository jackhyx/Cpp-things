class Solution {
vector<int> temp;
void surgeSort (vector<int>& nums, int l, int r) {
    if (l >= r) return ;
    int mid = l + (r - l) / 2;
    surgeSort(nums, l, mid);
    surgeSort(nums, mid + 1, r);

    int i = l, j = mid + 1, pos = l;
    while (i <= mid && j <= r) {
        if (nums[i] < nums[j]) {
            temp[pos] = nums[i];
            ++i;
        } else {
            temp[pos] = nums[j];
            ++j;
        }
        ++pos;
    }
    for (int k = i; k <= mid; ++k) {
        temp[pos++] = nums[k]; 
    }
    for (int k = j; k <= r; ++k) {
        temp[pos++] = nums[k];
    }
    copy(temp.begin() + l, temp.begin() + r + 1, nums.begin() + l);
}
public:
vector<int> sortArray(vector<int>& nums) {
int n = nums.size();
temp = nums;
surgeSort(nums, 0, n - 1);
return nums;

    }
};
